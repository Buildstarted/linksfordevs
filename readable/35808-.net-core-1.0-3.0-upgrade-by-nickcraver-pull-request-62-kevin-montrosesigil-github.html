To build (from solution root)

dotnet build
# To Run Tests (net461, netcoreapp2.2, netcoreapp3.0)
dotnet test
# To Generate NuGet
dotnet pack .\src\Sigil\Sigil.csproj -c Release /p:PackageOutputPath=..\..\.nupkgs /p:CI=true
