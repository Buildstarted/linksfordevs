Step 1: It starts with your SQL

  CREATE PROCEDURE dbo.FeedbackInsert
  (
      @FeedbackId int out,
      @FirstName nvarchar(32),
      @LastName nvarchar(32),
      @Email varchar(64),
      @Subject nvarchar(32),
      @Message nvarchar(1024)
  )
  AS
  BEGIN

      SET NOCOUNT ON;

      INSERT INTO [dbo].[Feedback]
      (
          LastName,
          FirstName,
          Email,
          Subject,
          Message,
          Created
      )
      VALUES
      (
          @LastName,
          @FirstName,
          @Email,
          @Subject,
          @Message,
          SYSUTCDATETIME()
      );

      SET @FeedbackId = SCOPE_IDENTITY();

      RETURN 1;

  END;
