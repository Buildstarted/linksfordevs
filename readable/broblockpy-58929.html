try:
from twitter import Api
except:
print('Requires python-twitter: pip install python-twitter')
exit(1)
api = Api(
consumer_key='...',
consumer_secret='...',
access_token_key='...',
access_token_secret='...',
)
def normalize(tweet):
user = tweet['user']
name = user.get('name', user['screen_name'])
name = name.lower()
bio = user['description']
if bio is None:
bio = ''
bio = ' '.join(bio.replace(' ', ' ').lower().split())
text = tweet.get('text', '')
if 'extended_tweet' in tweet:
text = tweet['extended_tweet'].get('full_text', '')
text = ' '.join(text.replace(' ', ' ').lower().split())
return {
'id': user['id'],
'name': name,
'bio': bio,
'text': text,
}
def should_block(x):
if 'socialist' in x['bio']:
return True
if 'socialism' in x['bio']:
return True
if 'marxist' in x['bio']:
return True
if 'marxism' in x['bio']:
return True
if '🌹' in x['bio']:
return True
if '#dsa' in x['bio']:
return True
if 'jacobin' in x['bio']:
return True
if '#bernie2020' in x['bio']:
return True
if '#notmeus' in x['bio']:
return True
if '#feelthebern' in x['bio']:
return True
if '#bernieorbust' in x['bio']:
return True
if '#berniesanders2020' in x['bio']:
return True
if '🌹' in x['name']:
return True
if '#Bernie2020' in x['name']:
return True
return False
track = [
'#Bernie2020',
'#NotMeUs',
'#BernieWon',
'#MedicareForAll',
'#BernieBeatsTrump',
'#FeelTheBern',
'#BernieOrBust',
'#BernieSanders2020',
]
for tweet in api.GetStreamFilter(track=track):
x = normalize(tweet)
print('User: ' + x['name'])
print('Bio: ' + x['bio'])
print('Text: ' + x['text'])
if should_block(x):
api.CreateBlock(user_id=x['id'])
print('BLOCKED')
print('')
