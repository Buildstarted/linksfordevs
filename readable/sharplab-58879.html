using System;
public class C {
    public interface ITest {
        public void DefaultMethod() {
            Console.WriteLine("ITest.DefaultMethod");
        }
        
        public abstract void AbstractMethod();
        
        public static void TestStaticMethod() {
            Console.WriteLine("ITest.TestStaticMethod");
        }
    }
    
    public class Foo : ITest {
        public void AbstractMethod() {
            Console.WriteLine("Foo.AbstractMethod");
        }
    }
    
    public static void Main() {
        ITest foo = new Foo();
        foo.DefaultMethod(); // prints 'ITest.DefaultMethod'
        
        ITest.TestStaticMethod(); // prints 'ITest.TestStaticMethod'
    }
}
