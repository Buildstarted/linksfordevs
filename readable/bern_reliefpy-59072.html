def normalize(tweet):
user = tweet['user']
name = user.get('name', user['screen_name'])
bio = user['description']
if bio is None:
bio = ''
bio = ' '.join(bio.replace(' ', ' ').split())
url = user['url']
if url is None:
url = ''
return {
'id': user['id'],
'name': name.lower(),
'handle': user['screen_name'],
'bio': bio.lower(),
'url': url.lower(),
}
def should_block(x):
if any(term in x['bio'] for term in BLOCK_BIO):
return True
if any(term in x['url'] for term in BLOCK_URL):
return True
if any(term in x['name'] for term in BLOCK_NAME):
return True
return False

if should_block(x):
